#include <SoftwareSerial.h>

// Recebe e envia dados
SoftwareSerial bluetooth(3, 2);

// Motor Esquerdo
// ESQUERDO FRENTE
#define PINO_1 5
// ESQUERDO TRÁS
#define PINO_2 6

// Motor Direito
// DIREITO FRENTE
#define PINO_3 9
// DIREITO TRÁS
#define PINO_4 10

// Velocidade Padrão
const int VELOCIDADE = 200;

void setup()
{
  // Saídas
  pinMode(PINO_1, OUTPUT);
  pinMode(PINO_2, OUTPUT);
  pinMode(PINO_3, OUTPUT);
  pinMode(PINO_4, OUTPUT);

  // Comunicações
  Serial.begin(9600);
  bluetooth.begin(9600);

  Serial.println("Carrinho pronto!!!");

  // Função de parada
  parar();
}

//////////////////////////  DIREÇÕES BÁSICAS  //////////////////////////

void parar()
{
  analogWrite(PINO_1, 0);
  analogWrite(PINO_2, 0);
  analogWrite(PINO_3, 0);
  analogWrite(PINO_4, 0);
}

void frente()
{
  analogWrite(PINO_1, LOW);
  analogWrite(PINO_2, VELOCIDADE);
  analogWrite(PINO_3, VELOCIDADE);
  analogWrite(PINO_4, LOW);
}

void tras()
{
  analogWrite(PINO_1, VELOCIDADE);
  analogWrite(PINO_2, LOW);
  analogWrite(PINO_3, LOW);
  analogWrite(PINO_4, VELOCIDADE);
}

void direita()
{
  analogWrite(PINO_1, VELOCIDADE);
  analogWrite(PINO_2, LOW);
  analogWrite(PINO_3, VELOCIDADE);
  analogWrite(PINO_4, LOW);
}

void esquerda()
{
  analogWrite(PINO_1, LOW);
  analogWrite(PINO_2, VELOCIDADE);
  analogWrite(PINO_3, LOW);
  analogWrite(PINO_4, VELOCIDADE);
}


//////////////////////////  ATAQUES ESPECIAIS  //////////////////////////

// ATAQUE: GIRO 180°
void ataqueGiro180()
{
  // GIRO
  analogWrite(PINO_1, LOW); 
  analogWrite(PINO_2, 255);
  analogWrite(PINO_3, LOW);
  analogWrite(PINO_4, 255);
  
  delay(300); 
  
  parar();
}

// ATAQUE: DASH
void ataqueDash()
{
  // DASH
  analogWrite(PINO_1, LOW);
  analogWrite(PINO_2, 255);
  analogWrite(PINO_3, 255);
  analogWrite(PINO_4, LOW);
  
  delay(800);
  
  parar();
}

// ATAQUE: DRIBLE ZIGZAGUE
void ataqueZigZag()
{
  // VIRA PRA ESQUERDA
  analogWrite(PINO_1, LOW);
  analogWrite(PINO_2, 90);
  analogWrite(PINO_3, 255);
  analogWrite(PINO_4, LOW);
  delay(800);
  
  // VIRA PRA DIREITA
  analogWrite(PINO_1, LOW);
  analogWrite(PINO_2, 255);
  analogWrite(PINO_3, 120);
  analogWrite(PINO_4, LOW);
  delay(800);
  
  // Passo 3: Volta para o centro e dá uma arrancada para frente
  frente();
  delay(500);
  
  parar();
}

//////////////////////////  LOOP PRINCIPAL  //////////////////////////

void loop()
{
  if (bluetooth.available() > 0)
  {
    char comando = bluetooth.read();
    
    Serial.print("Comando recebido: ");
    Serial.println(comando);

    //////////////////////////  MOVIMENTAÇÃO PADRÃO  //////////////////////////

    if (comando == 'F' || comando == 'f') { frente(); }
    if (comando == 'B' || comando == 'b') { tras(); }
    if (comando == 'L' || comando == 'l') { esquerda(); }
    if (comando == 'R' || comando == 'r') { direita(); }
    if (comando == 'S' || comando == 's') { parar(); }

    //////////////////////////  GATILHOS DOS ESPECIAIS  //////////////////////////

    if (comando == 'X' || comando == 'x') 
    {
      ataqueGiro180();
    }
    if (comando == 'Y' || comando == 'y') 
    {
      ataqueDash();
    }
    if (comando == 'Z' || comando == 'z') 
    {
      ataqueZigZag();
    }
  }else{
    // Serial.println("Não leu");
  }
}
